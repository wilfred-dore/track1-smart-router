\begin{frame}{The problem: accuracy is a gate, tokens are the score}
  \begin{columns}[T]
    \column{0.55\textwidth}
      \textbf{Enterprise reality}
      \begin{itemize}
        \item Not every task deserves a premium model
        \item Host cheap local models in-house, call the paid API only when it matters
        \item Track 1 scores exactly that: pass the 80\% accuracy gate, then rank by \textbf{fewest Fireworks tokens}
      \end{itemize}
    \column{0.4\textwidth}
      \textbf{Consequences for design}
      \begin{itemize}
        \item Local inference $=$ 0 tokens $\Rightarrow$ use it aggressively
        \item Every escalation must be \emph{deliberate}
        \item Route by \textbf{verifiability}, not difficulty
      \end{itemize}
  \end{columns}
  \vspace{0.4cm}
  \centering
  \small 8 task categories: factual QA $\cdot$ math $\cdot$ sentiment $\cdot$ summarization $\cdot$ NER $\cdot$ code debugging $\cdot$ logic $\cdot$ code generation
\end{frame}

\begin{frame}{Measured, not assumed: the descent from 6{,}559 to 639 tokens}
  \centering
  \begin{tabular}{llrl}
    \toprule
    \textbf{Step} & \textbf{Change} & \textbf{Tokens} & \textbf{Accuracy} \\
    \midrule
    v11 & everything escalated, 18 calls & 6{,}559 & 94.7\% (scored) \\
    +batch & escalations grouped: one thinking session & 3{,}510 & 19/19 local eval \\
    +local & summarization \& code answered on-device & 2{,}758 & 19/19 \\
    +\texttt{reasoning\_effort=none} & billed thinking switched off & 548 & 19/19 \\
    +prompt trim, +solo logic & final profile (v20) & \textbf{639--649} & \textbf{19/19 $\times$4 runs} \\
    \bottomrule
  \end{tabular}
  \vspace{0.4cm}
  \begin{itemize}
    \item Every step measured on the \textbf{real Fireworks API} with the real \texttt{kimi-k2p7-code}
    \item Batching suppresses billed reasoning ($-74\%$ measured in the literature; we observed the same)
    \item Batched logic without thinking is nondeterministically wrong $\rightarrow$ logic runs solo, $\sim$6 tokens per task
  \end{itemize}
\end{frame}

\begin{frame}{Validated three ways before ever resubmitting}
  \begin{itemize}
    \item \textbf{Real-API evaluation}: 19 practice-style fixtures, 19/19 on four consecutive runs, 639--649 tokens (near-deterministic)
    \item \textbf{Generalization gauntlet}: 19 fresh adversarial tasks written to break the pipeline --- caught a sentiment-lexicon trap, a semantic code-debug miss and NER label errors; all fixed, 18/19 (the last miss is genuinely ambiguous)
    \item \textbf{Independent LLM-judge simulation}: a different model family graded all 38 answers the way the official judge would --- caught nonsense justifications a regex checker cannot see
    \item \textbf{CI replays the grading environment exactly} (linux/amd64, 4\,GB RAM, 2 vCPU): 19 tasks in 90\,s, 1.97\,GB compressed image, schema and size gates before any publish
    \item Everything config-driven; no hardcoded answers, no caching across runs; the only runtime network call is \texttt{FIREWORKS\_BASE\_URL}
  \end{itemize}
\end{frame}

\begin{frame}{Takeaways}
  \begin{itemize}
    \item \textbf{Zero-cost first}: solvers + a bundled 3B absorb everything that can be verified for free; the paid API sees only the hard residue
    \item \textbf{Route by reliability, not difficulty}: categories where small models fail \emph{silently} escalate outright
    \item \textbf{Measure everything}: stand-in proxies before credits, real-API sweeps after, an adversarial gauntlet and a simulated judge before each freeze
    \item \textbf{Plumbing wins hackathons}: most failed submissions break on infrastructure --- ours is CI-proven against the exact grading limits
  \end{itemize}
